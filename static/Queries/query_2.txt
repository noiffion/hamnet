SELECT reviews.review_title, reviews.review_link, reviews.performance_date,
       theatres.webpage, theatres.theatre_name, cities.city_name, 
       performances.user_id, users.username, performances.review_id, 
       performances.id
FROM performances JOIN reviews 
ON performances.review_id = reviews.id
JOIN (theatres JOIN cities ON theatres.city_id = cities.id) 
ON performances.theatre_id = theatres.id
JOIN users 
ON performances.user_id = users.id
WHERE performances.play_id = {0} AND performances.user_id = {1}
ORDER BY reviews.performance_date DESC;
